##02.01移除重复节点## </br>
Python中操作链表，使用指针指代一个节点时分一下两种情况： </br>
1. pos = pos.next  -> 仅仅只是pos指向的位置向后移一个，不改变pos的值 </br>
2. pos.next = pos.next.nex  -> 对除pos之外的元素进行更改时会改变该值，注意！！！ </br>
```python
    def removeDuplicateNodes(self, head: ListNode) -> ListNode:
        if not head:
            return head
        s = {head.val}
        pos = head
        while pos.next:
            cur = pos.next
            if cur.val not in s:
                s.add(cur.val)
                pos = pos.next
            else:
                pos.next = pos.next.next
        return head
```

##2.两数相加## </br>
涉及到链表的题都可以加一个伪节点dummyNode，相加的话一般设一个carry
```python
class Solution:
    def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode:
        ptr1, ptr2 = l1, l2
        
        dummyNode = ListNode(-1)
        cur = dummyNode
        carry = 0
        while ptr1 or ptr2:
            val1 = ptr1.val if ptr1 else 0
            val2 = ptr2.val if ptr2 else 0
            val = (val1 + val2 + carry) % 10
            carry = (val1 + val2 + carry) // 10
            node = ListNode(val)
            cur.next = node
            cur = node
            if ptr1: ptr1 = ptr1.next
            if ptr2: ptr2 = ptr2.next
        if carry == 1:
            node = ListNode(1)
            cur.next = node
            cur = node
        return dummyNode.next
```

##19. 删除链表的倒数第 N 个结点## </br>
涉及到链表的题都可以加一个伪节点dummyNode，注意边界条件，fast从head开始，slow从dummy开始，因为要找到目的节点的前一个节点
```python
class Solution:
    def removeNthFromEnd(self, head: ListNode, n: int) -> ListNode:
        dummyNode = ListNode()
        dummyNode.next = head
        fast = head
        for _ in range(n):
            fast = fast.next
        slow = dummyNode
        while fast:
            fast = fast.next
            slow = slow.next
        slow.next = slow.next.next
        return dummyNode.next
```

##21. 合并两个有序链表## </br>
第一种做法，迭代
```python
class Solution:
    def mergeTwoLists(self, l1: ListNode, l2: ListNode) -> ListNode:
        dummyNode = ListNode()
        cur = dummyNode
        while l1 and l2:
            if l1.val < l2.val:
                cur.next = l1
                l1 = l1.next
            else:
                cur.next = l2
                l2 = l2.next
            cur = cur.next
        if l1:
            cur.next = l1
        if l2:
            cur.next = l2
        return dummyNode.next
```
第二种做法，递归
```python
class Solution:
    def mergeTwoLists(self, l1: ListNode, l2: ListNode) -> ListNode:
        if not l1:
            return l2
        if not l2:
            return l1
        
        if l1.val < l2.val:
            l1.next = self.mergeTwoLists(l1.next, l2)
            return l1
        else:
            l2.next = self.mergeTwoLists(l1, l2.next)
            return l2
```

##23. 合并K个升序链表## </br>
第一种做法，使用堆和哈希表来记录节点，速度最快
```python
class Solution:
    def mergeKLists(self, lists: List[ListNode]) -> ListNode:
        hq = []
        for i, listNode in enumerate(lists):
            if listNode:
                hq.append((listNode.val, i))
        
        dummyNode = ListNode()
        cur = dummyNode
        heapq.heapify(hq)
        while hq:
            val, idx = heapq.heappop(hq)
            node = lists[idx]
            cur.next = node
            node = node.next
            lists[idx] = node
            if node:
                heapq.heappush(hq, (node.val, idx))
                
            cur = cur.next
        
        return dummyNode.next
```
第二种做法，分而治之
```python
class Solution:
    def mergeKLists(self, lists: List[ListNode]) -> ListNode:
        if not lists:
            return None
        n = len(lists)
        return self.merge(lists, 0, n - 1)

    def merge(self, lists, left, right):
        if left >= right:
            return lists[left]
        mid = left + (right - left) // 2
        l1 = self.merge(lists, left, mid)
        l2 = self.merge(lists, mid + 1, right)
        return self.mergeTwoLists(l1, l2)
    
    def mergeTwoLists(self, l1, l2):
        if not l1: return l2
        if not l2: return l1
        if l1.val < l2.val:
            l1.next = self.mergeTwoLists(l1.next, l2)
            return l1
        else:
            l2.next = self.mergeTwoLists(l1, l2.next)
            return l2
```
